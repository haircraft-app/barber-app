  <script>
    const SUPABASE_URL = "https://hnvqauxqxtdzgyqumkqa.supabase.co";
    const SUPABASE_KEY = "sb_publishable_k1pGTW92yAbLX3PKRQjFpQ_0AQBRker";;

    var clients = [];
    var currentClientId = null;
    var formMode = "add";

    function supabaseHeaders(extra) {
      var headers = {
        "apikey": SUPABASE_KEY,
        "Authorization": "Bearer " + SUPABASE_KEY,
        "Content-Type": "application/json"
      };

      if (extra) {
        Object.keys(extra).forEach(function(k) {
          headers[k] = extra[k];
        });
      }

      return headers;
    }

    async function loadClientsFromSupabase() {
      var box = document.getElementById("clientList");
      box.innerHTML = '<div class="empty">Se încarcă...</div>';

      try {
        var response = await fetch(
          SUPABASE_URL + "/rest/v1/clienti?select=id,nume,telefon&order=nume.asc",
          {
            method: "GET",
            headers: supabaseHeaders()
          }
        );

        if (!response.ok) {
          throw new Error("Eroare Supabase: " + response.status);
        }

        var data = await response.json();

        console.log(data);

        clients = data.map(function(c) {
          return {
            id: c.id,
            name: c.nume || "",
            phone: c.telefon || "",
            email: "",
            birthday: "",
            notes: ""
          };
        });

        renderClients();

      } catch (err) {
        box.innerHTML = '<div class="empty">Nu pot încărca din Supabase. Verifică URL și KEY.</div>';
        console.log(err);
      }
    }

    function showScreen(id) {
      var screens = document.querySelectorAll(".screen");
      screens.forEach(function(s) {
        s.classList.remove("active");
      });

      document.getElementById(id).classList.add("active");
    }

    function renderClients() {
      document.getElementById("totalClienti").innerText = clients.length;
      var box = document.getElementById("clientList");
      var q = document.getElementById("searchInput").value.toLowerCase().trim();

      var filtered = clients
        .filter(function(c) {
          return c.name.toLowerCase().indexOf(q) !== -1 ||
                 c.phone.toLowerCase().indexOf(q) !== -1;
        })
        .sort(function(a, b) {
          return a.name.localeCompare(b.name);
        });

      if (filtered.length === 0) {
        box.innerHTML = '<div class="empty">Nu există clienți găsiți.</div>';
        return;
      }

      box.innerHTML = filtered.map(function(c) {
        return '' +
          '<div class="client-card" onclick="openDetails(\'' + c.id + '\')">' +
            '<div class="icon-circle">' +
              '<svg viewBox="0 0 24 24">' +
                '<circle cx="12" cy="8" r="4"></circle>' +
                '<path d="M4 20c0-4 4-6 8-6s8 2 8 6"></path>' +
              '</svg>' +
            '</div>' +
            '<div class="client-info">' +
              '<div class="client-name">' + escapeHtml(c.name) + '</div>' +
              '<div class="client-phone">' + escapeHtml(c.phone) + '</div>' +
            '</div>' +
            '<div class="arrow">›</div>' +
          '</div>';
      }).join("");
    }

    function openDetails(id) {
      currentClientId = id;

      var c = clients.find(function(x) {
        return String(x.id) === String(id);
      });

      if (!c) return;

      document.getElementById("detailName").textContent = c.name;
      document.getElementById("detailPhone").textContent = c.phone || "Fără telefon";
      document.getElementById("detailEmail").textContent = "Email neactivat în test";
      document.getElementById("detailBirthday").textContent = "Data naștere neactivată în test";

      showScreen("screenDetails");
    }

    function openAdd() {
      formMode = "add";
      currentClientId = null;

      document.getElementById("formTitle").textContent = "Adaugă client";
      document.getElementById("formName").value = "";
      document.getElementById("formPhone").value = "";
      document.getElementById("formEmail").value = "";
      document.getElementById("formBirthday").value = "";
      document.getElementById("formNotes").value = "";
      document.getElementById("formDeleteBtn").style.display = "none";

      showScreen("screenForm");
    }

    function openEdit() {
      var c = clients.find(function(x) {
        return String(x.id) === String(currentClientId);
      });

      if (!c) return;

      formMode = "edit";

      document.getElementById("formTitle").textContent = "Editează client";
      document.getElementById("formName").value = c.name || "";
      document.getElementById("formPhone").value = c.phone || "";
      document.getElementById("formEmail").value = "";
      document.getElementById("formBirthday").value = "";
      document.getElementById("formNotes").value = "";
      document.getElementById("formDeleteBtn").style.display = "block";

      showScreen("screenForm");
    }

    async function saveClient() {
      var name = document.getElementById("formName").value.trim();
      var phone = document.getElementById("formPhone").value.trim();
      var email = document.getElementById("formEmail").value.trim();
      var birthday = document.getElementById("formBirthday").value;
      var notes = document.getElementById("formNotes").value.trim();


      if (!name) {
        alert("Completează numele clientului.");
        return;
      }

      if (!phone) {
        alert("Completează telefonul clientului.");
        return;
      }

      try {
        if (formMode === "add") {
          var responseAdd = await fetch(
            SUPABASE_URL + "/rest/v1/clienti?select=id,nume,telefon,email,data_nastere,notite",
            {
              method: "POST",
              headers: supabaseHeaders({
                "Prefer": "return=representation",
                "Content-Type": "application/json"
              }),
              body: JSON.stringify({
                nume: name,
                telefon: phone,
                email: email,
                data_nastere: birthday || null,
                notite: notes
              })
            }
          );

          if (!responseAdd.ok) {
            throw new Error("Eroare la adăugare: " + responseAdd.status);
          }

          var inserted = await responseAdd.json();
          currentClientId = inserted[0].id;

        } else {
          var responseEdit = await fetch(
            SUPABASE_URL + "/rest/v1/clienti?id=eq." + encodeURIComponent(currentClientId),
            {
              method: "PATCH",
              headers: supabaseHeaders(),
              body: JSON.stringify({
                nume: name,
                telefon: phone,
                email: email,
                data_nastere: birthday,
                notite: notes
              })
            }
          );

          if (!responseEdit.ok) {
            throw new Error("Eroare la editare: " + responseEdit.status);
          }
        }

        await loadClientsFromSupabase();
        openDetails(currentClientId);

      } catch (err) {
        alert("Nu s-a putut salva în Supabase. Verifică URL, KEY și RLS.");
        console.log(err);
      }
    }

    async function deleteCurrentClient() {
      if (!currentClientId) return;

      var c = clients.find(function(x) {
        return String(x.id) === String(currentClientId);
      });

      if (!c) return;

      if (!confirm("Ștergi clientul " + c.name + "?")) return;

      try {
        var response = await fetch(
          SUPABASE_URL + "/rest/v1/clienti?id=eq." + encodeURIComponent(currentClientId),
          {
            method: "DELETE",
            headers: supabaseHeaders()
          }
        );

        if (!response.ok) {
          throw new Error("Eroare la ștergere: " + response.status);
        }

        currentClientId = null;
        await loadClientsFromSupabase();
        showScreen("screenList");

      } catch (err) {
        alert("Nu s-a putut șterge clientul din Supabase.");
        console.log(err);
      }
    }

    function programClient() {
      alert("Aici se va lega cu pagina Programare.");
    }

    function clientHistory() {
      alert("Aici se va afișa istoricul programărilor.");
    }

    function goBack() {
      var detailsActive = document.getElementById("screenDetails").classList.contains("active");
      var formActive = document.getElementById("screenForm").classList.contains("active");

      if (detailsActive) {
        showScreen("screenList");
        return;
      }

      if (formActive) {
        if (formMode === "edit" && currentClientId) {
          openDetails(currentClientId);
        } else {
          showScreen("screenList");
        }
        return;
      }

      alert("Înapoi la pagina principală.");
    }

    function escapeHtml(text) {
      return String(text || "")
        .replaceAll("&", "&amp;")
        .replaceAll("<", "&lt;")
        .replaceAll(">", "&gt;")
        .replaceAll('"', "&quot;")
        .replaceAll("'", "&#039;");
    }
